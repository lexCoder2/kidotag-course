Un **middleware** es simplemente una función con la firma `(req, res, next)`. Express ejecuta los middlewares en cadena — el orden importa muchísimo.

## El patrón middleware

```
Petición → [middleware 1] → [middleware 2] → [middleware N] → Ruta → Respuesta
             ↓ next()         ↓ next()         ↓ next()
```

```javascript
// Anatomía de un middleware
function miMiddleware(req, res, next) {
  // 1. Puede leer/modificar req y res
  // 2. Puede terminar la petición (res.json, res.send, etc.)
  // 3. O pasar al siguiente con next()
  // 4. O señalar un error con next(error)
  next(); // ← continúa la cadena
}
```

const app = express();

// ─── 1. Middleware de aplicación (global) ───────
app.use(express.json());

// Logger personalizado
app.use((req, res, next) => {
  const inicio = Date.now();
  res.on("finish", () => {
    const ms = Date.now() - inicio;
    console.log(\`\${req.method} \${req.url} → \${res.statusCode} [\${ms}ms]\`);
  });
  next();
});

// ─── 2. Middleware de ruta (solo en /api/alumnos) ──
function verificarToken(req, res, next) {
  const auth = req.headers.authorization;
  if (!auth || !auth.startsWith("Bearer ")) {
    return res.status(401).json({ ok: false, error: "Token requerido" });
  }
  const token = auth.split(" ")[1];
  // En kidotag10: jwt.verify(token, process.env.JWT_SECRET)
  if (token === "token-valido") {
    req.usuario = { id: "1", rol: "profesor" };
    next();
  } else {
    res.status(403).json({ ok: false, error: "Token inválido" });
  }
}

// ─── 3. Rutas ────────────────────────────────────
app.get("/publico", (req, res) => {
  res.json({ mensaje: "Esta ruta no necesita token" });
});

app.get("/api/alumnos", verificarToken, (req, res) => {
  res.json({
    ok: true,
    usuario: req.usuario,
    data: [{ id: 1, nombre: "Ana García" }],
  });
});

// ─── 4. Middleware de error (4 parámetros!) ──────
app.use((err, req, res, next) => {
  console.error("Error:", err.message);
  res.status(500).json({ ok: false, error: err.message });
});

app.listen(3001, () => console.log("Servidor en puerto 3001"));

// Prueba:
// GET /publico → funciona sin token
// GET /api/alumnos → 401 sin token
// GET /api/alumnos con header "Authorization: Bearer token-valido" → OK`,

## CORS — Cross-Origin Resource Sharing

```javascript
const cors = require("cors");

// En kidotag10: solo permite el origen del frontend
app.use(
  cors({
    origin: process.env.FRONTEND_URL || "http://localhost:5173",
    credentials: true, // Permite cookies
    methods: ["GET", "POST", "PUT", "DELETE", "OPTIONS"],
  }),
);
```

<Callout variante="warning" titulo="El orden de app.use() importa">
  Los middlewares se ejecutan en el **orden en que se registran**. Siempre
  registra: 1. `cors()` primero 2. Luego `express.json()` 3. Después tus rutas
  4. El handler de errores **al final** (siempre)
</Callout>
