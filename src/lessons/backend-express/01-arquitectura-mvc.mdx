# Arquitectura MVC en kidotag10

kidotag10 usa el patrón **MVC** (Model-View-Controller) para organizar el código del backend. Cada parte tiene una responsabilidad clara.

## Las 4 capas de la API

```
src/
├── routes/          → ¿QUÉ URL responde a qué?
│   ├── alumno.routes.js
│   └── auth.routes.js
│
├── controllers/     → ¿CÓMO procesar la petición?
│   ├── alumno.controller.js
│   └── auth.controller.js
│
├── models/          → ¿QUÉ estructura tienen los datos?
│   ├── alumno.model.js
│   └── grupo.model.js
│
└── middlewares/     → ¿QUÉ validar antes de llegar al controller?
    └── auth.middleware.js
```

## Rutas: solo el mapa de URLs

```javascript
// src/routes/alumno.routes.js
const router = require("express").Router();
const {
  getAlumnos,
  crearAlumno,
  getAlumno,
  actualizarAlumno,
} = require("../controllers/alumno.controller");
const {
  verificarToken,
  esProfesor,
  esAdmin,
} = require("../middlewares/auth.middleware");

// GET /api/alumnos
router.get("/", verificarToken, esProfesor, getAlumnos);

// POST /api/alumnos
router.post("/", verificarToken, esAdmin, crearAlumno);

// GET /api/alumnos/:id
router.get("/:id", verificarToken, esProfesor, getAlumno);

// PUT /api/alumnos/:id
router.put("/:id", verificarToken, esAdmin, actualizarAlumno);

module.exports = router;
```

## Controladores: la lógica de negocio

```javascript
// src/controllers/alumno.controller.js

const getAlumnos = async (req, res) => {
  try {
    const { grado, activo } = req.query;
    const filtro = {};
    if (grado) filtro.grado = grado;
    if (activo !== undefined) filtro.activo = activo === "true";

    const alumnos = await Alumno.find(filtro)
      .populate("grupo", "nombre grado") // Join con Grupo
      .sort({ nombre: 1 });

    res.json({ ok: true, data: alumnos });
  } catch (error) {
    res.status(500).json({ ok: false, error: error.message });
  }
};

const crearAlumno = async (req, res) => {
  try {
    const { nombre, matricula, grupoId } = req.body;

    // Verificar matrícula única
    const existe = await Alumno.findOne({ matricula });
    if (existe) {
      return res
        .status(409)
        .json({ ok: false, error: "Matrícula ya registrada" });
    }

    const alumno = await Alumno.create({ nombre, matricula, grupo: grupoId });
    res.status(201).json({ ok: true, data: alumno });
  } catch (error) {
    if (error.name === "ValidationError") {
      return res.status(400).json({ ok: false, error: error.message });
    }
    res.status(500).json({ ok: false, error: error.message });
  }
};
```
