**Express** es el framework web más popular para Node.js. Añade una capa sencilla sobre el módulo HTTP nativo para manejar rutas, peticiones y respuestas de forma elegante.

## Sin Express vs con Express

```javascript
// ❌ Node.js puro — mucho código para algo simple
const http = require("http");
const server = http.createServer((req, res) => {
  if (req.url === "/alumnos" && req.method === "GET") {
    res.writeHead(200, { "Content-Type": "application/json" });
    res.end(JSON.stringify([{ id: 1, nombre: "Ana" }]));
  }
});
server.listen(3000);

// ✅ Con Express — limpio y expresivo
const express = require("express");
const app = express();

app.get("/alumnos", (req, res) => {
  res.json([{ id: 1, nombre: "Ana" }]);
});

app.listen(3000);
```

## El objeto `req` y `res`

```javascript
app.post("/ruta", (req, res) => {
  // ── req (Request) — lo que LLEGA ──
  req.body; // Cuerpo JSON o form-data
  req.params.id; // Segmentos de ruta: /ruta/:id
  req.query.page; // Query strings: /ruta?page=2
  req.headers["authorization"]; // Headers HTTP
  req.method; // 'GET', 'POST', 'PUT', 'DELETE'

  // ── res (Response) — lo que ENVÍAS ──
  res.json({ ok: true, data }); // JSON con status 200
  res.status(201).json({ ok: true }); // JSON con status 201
  res.status(400).json({ ok: false, error: "Bad request" });
  res.send("texto plano");
  res.sendFile(path.join(__dirname, "index.html"));
});
```
