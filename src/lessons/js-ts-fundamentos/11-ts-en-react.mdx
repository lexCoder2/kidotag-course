# TypeScript en React: props y estado tipado

En React, TypeScript ayuda mucho con props y estado.

## Props tipadas

```ts
type CardProps = {
  nombre: string;
  presente: boolean;
};

function Card({ nombre, presente }: CardProps) {
  return <div>{nombre} - {presente ? "OK" : "Falta"}</div>;
}
```

## useState tipado

```ts
const [total, setTotal] = useState<number>(0);
```

// Funcion pura — recibe number, devuelve el tipo union
function getEstado(p: number): EstadoAsistencia {
  if (p >= 90) return "Excelente";
  if (p >= 75) return "Regular";
  return "En riesgo";

// Props tipadas — TypeScript valida que se pasen correctamente
type TarjetaProps = { alumno: Alumno };

function TarjetaAlumno({ alumno }: TarjetaProps) {
  const estado = getEstado(alumno.asistencia);
  const bg = estado === "Excelente" ? "#dcfce7" : estado === "Regular" ? "#fef9c3" : "#fee2e2";
  return (

        {alumno.asistencia}% — {estado}

    </div>
  );
}

// useState<T> — T define que tipo puede guardarse en el estado
export default function App() {
