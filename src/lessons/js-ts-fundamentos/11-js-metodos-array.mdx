# map, filter y reduce con ejemplos reales

Estos metodos se usan todo el tiempo en React.

- `map`: transformar cada elemento.
- `filter`: quedarse solo con algunos.
- `reduce`: resumir en un valor final.

```javascript
const nums = [1, 2, 3];
const dobles = nums.map((n) => n * 2); // [2,4,6]
```

import { getEstado } from "./asistencia.js";
import { formatNombre, formatPorcentaje } from "./helpers.js";

// map: transforma cada elemento — devuelve solo los nombres formateados
export function obtenerNombres(alumnos) {
  return alumnos.map(function (a) {
    return formatNombre(a.nombre);
  });
}

// filter: selecciona elementos — solo los alumnos en riesgo (<75%)
export function filtrarEnRiesgo(alumnos) {
  return alumnos.filter(function (a) {
    return a.asistencia < 75;
  });
}

// reduce: resume en un valor — suma todas las faltas
export function totalFaltas(alumnos) {
  return alumnos.reduce(function (acc, a) {
    return acc + a.faltas;
  }, 0);
}

// Combina los tres en un resumen del grupo
export function resumenGrupo(alumnos) {
  var tf = totalFaltas(alumnos);
  return {
    total: alumnos.length,
    enRiesgo: filtrarEnRiesgo(alumnos).length,
    totalFaltas: tf,
    promedioFaltas: (tf / alumnos.length).toFixed(1),
    nombres: obtenerNombres(alumnos),
  };
}`,
